function copyLink(btn, inputId) { const input = document.getElementById(inputId); if (!input) return; navigator.clipboard.writeText(input.value).then(() => { const original = btn.textContent; btn.textContent = 'Copied'; setTimeout(() => { btn.textContent = original; }, 1500); }); } // Auto-dismiss flash messages document.addEventListener('DOMContentLoaded', () => { document.querySelectorAll('.flash').forEach((el) => { setTimeout(() => { el.style.transition = 'opacity 0.4s ease, transform 0.4s ease'; el.style.opacity = '0'; el.style.transform = 'translateX(8px)'; setTimeout(() => el.remove(), 400); }, 4000); }); });
